
=======  IN vs OUT example

CREATE OR REPLACE PROCEDURE query_empinfo
 (p_empid    IN hr_employees.employee_id%TYPE,
  p_lname  OUT hr_employees.last_name%TYPE,
  p_salary OUT hr_employees.salary%TYPE) IS
BEGIN
    SELECT last_name, salary INTO p_lname, p_salary
    FROM   hr_employees
    WHERE  employee_id = p_empid;
END  query_empinfo;
/



Test (Executing)

DECLARE
 v_emp_lname char(50);
 v_emp_sal number(9,2);
BEGIN
